Tic Tac Toe

A clean, responsive Tic Tac Toe game with a focus on smooth interactions and simple logic. 

// Tic Tac Toe (excerpt)
const winningLines = [
  [0,1,2],[3,4,5],[6,7,8],
  [0,3,6],[1,4,7],[2,5,8],
  [0,4,8],[2,4,6]
];

function calculateWinner(squares) {
  for (const [a,b,c] of winningLines) {
    if (squares[a] && squares[a] === squares[b] && squares[a] === squares[c]) {
      return squares[a];
    }
  }
  return null;
}

// Docs: Simple 3x3 grid game. Click a square to place X/O.
// First to align three wins. Includes reset button and status text.
